# XML

- eXtensible Markup Language의 약자이다.
- 확장 가능한 마크업 언어이다.
- 데이터를 관리하는 목적으로 사용된다.
- setup 을 하는 목적으로 사용된다. (.txt)
- 서버는 처리 후 xml을 리턴한다
- 사용자가 정의한 태그를 통해 데이터를 정의한다.

```xml
<이름>홍길동</이름>
<나이>26</나이>
<주소>주소</주소>

<이름>Parker</이름>
<나이>22</나이>
<주소>주소2</주소>
```

- 특정 태그를 정하고 데이터를 관리한다.



#### index.html


```html
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<title>Insert title here</title>
</head>
<body>

	<p id="demo">

	<script type="text/javascript">
  var xhttp = new XMLHttpRequest();
    xhttp.onreadystatechange = function() {
      if (this.readyState == 4 && this.status == 200) {
        /* myFunc(this); */
        // nodeFunc(this);
        dataFunc(this);
      }
    };
    xhttp.open("GET", "sample2.xml", true);
    xhttp.send();

    // 값 전부 가져오기
    function myFunc(xml) {
      var x, i, txt, xmlDoc;

      xmlDoc = xml.responseXML;
      txt = "";
      x = xmlDoc.getElementsByTagName('번호');

      for (i = 0; i < x.length; i++) {
        txt += x[i].childNodes[0].nodeValue + "<br>"
      }

      document.getElementById('demo').innerHTML = txt;
    }

    // node 전부 가져오기
    function nodeFunc(xml) {
      var x, i, xmlDoc, txt;

      xmlDoc = xml.responseXML;
      txt = "";
      x = xmlDoc.documentElement.childNodes;

      for (i = 0; i < x.length; i++) {
        if (x[i].nodeType == 1) {
          txt += x[i].nodeName + "<br>"
        }
      }
      document.getElementById('demo').innerHTML = txt;
    }

    // 데이터를 가져오는 방법
    function dataFunc(xml) {
      var x, i, xmlDoc, txt;

      xmlDoc = xml.responseXML;
      txt = "";				
      x = xmlDoc.getElementsByTagName('고객')[0];
      var xlen = x.childNodes.length;

      y = x.firstChild;

      for (var i = 0; i < xlen; i++) {
        if(y.nodeType == 1){
          txt += i + " " + y.nodeName + "<br>";
        }
        y = y.nextSibling;
      }
      document.getElementById('demo').innerHTML = txt;

    }
	</script>
</body>
</html>
```
